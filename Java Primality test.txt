import java.io.*;

    public static void main(String[] args) {
        Scanner sc =new Scanner(System.in);
        BigInteger n=sc.nextBigInteger();
        

        sc.close();
        boolean b=n.isProbablePrime(1);
        if(b){
            System.out.println("prime");
        }
        else{
            System.out.println("not prime");
        }
    }
}
